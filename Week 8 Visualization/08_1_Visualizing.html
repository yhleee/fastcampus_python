In [3]:
import matplotlib.pyplot as plt
plt.plot([1,2,3,4])
plt.ylabel('some numbers')
plt.show()

plot의 default 상황에서 기준 데이터는 y임. y축의 범위는 [1,2,3,4] 이므로 1.0 ~ 4.0의 범위를 갖는 그래프가 생성됨. x축의 데이터가 없을 경우, y축의 데이터에 기반하여 동일한 모양으로 추정해주는데, 0부터 시작하기 때문에 0.0 ~ 3.0의 범위를 가짐.

def f(t):
    return np.exp(-t) * np.cos(2*np.pi*t)

t2 = np.arange(0.0, 5.0, 0.02)

pie chart 그리기¶

In [89]:
import matplotlib.pyplot as plt
from matplotlib.gridspec import GridSpec

# 시각화할 데이터
labels = 'Frogs', 'Hogs', 'Dogs', 'Logs'
fracs1 = [15, 30, 45, 10]
fracs2 = [35, 10, 30, 25]

explode = (0, 0.25, 0, 0)

the_grid = GridSpec(2, 2)
plt.subplot(the_grid[1, 0], aspect=1)
plt.pie(fracs1, labels=labels, autopct='%1.1f%%', shadow=True)
plt.subplot(the_grid[0, 1], aspect=1)
plt.pie(fracs2, explode=explode, labels=labels, autopct='%.0f%%', shadow=True)

plt.show()
